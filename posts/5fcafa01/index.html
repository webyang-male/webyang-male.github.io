       Vue.component('item-one', {
           template: `
           <ul>
               <li>这是第一个组件</li>
               <li>这是第一个组件</li>
           </ul>

               <li>这是第二个组件</li>
           </ul>
           `
       })

           el: "#app1",

父子组件传值

<!-- 
       我们想要展示数据，所以诞生了我们的props
    -->
   <div id="app1">
       <!-- 你怎么传的值，就怎么在props里面接收，就怎么使用 -->
       <!-- 简写 -->
       <item-one :cuihua="msgs"></item-one>
       <!-- <item-one v-bind:cuihua="msgs"></item-one> -->
   </div>
   <script>
       Vue.component('item-one', {
           props: ['cuihua'], // 定义一个prop接受父组件的值
           template: `
           <ul>
               <li>{{cuihua}}</li>
               <li>{{cuihua}}</li>
           </ul>
           `
       })

       let vm1 = new Vue({
           el: "#app1",
           data: {
               msg: "像风一样自由",
               msgs: "像风一样自由werwerewrewr"
           }
       })
   </script>

父子组件传值(高级)

<div id="app1">
       <item-one
           v-bind:cuihua="msg"
           v-bind:goudan="msgs"
           v-for="item in list"
           v-bind:xiaohua="item"
           :key="item.id"
           :xiaoming="val"
           :xixi="change"
       ></item-one>
   </div>
   <script>
       Vue.component('item-one', {
           props: ['cuihua', "goudan", "xiaohua", "xiaoming", "xixi"], // 可以定义多个prop
           template: `<div>
               <p @click="xixi">{{xiaoming}}{{xiaohua.text}}{{cuihua}}{{goudan}}</p>
           </div>`
       })

       let vm1 = new Vue({
           el: "#app1",
           data: {
               msg: "像风一样自由",
               msgs: "像风一样自由的女子",
               list: [
                   {id: 0, text: '榴莲'},

                   {id: 2, text: '哈密瓜'},
                   {id: 3, text: '西瓜'}
               ]
           },
           computed: {
               val() {
                   return 520
               }
           },
           methods: {
               change() {
                   console.log('Hello Vue!')
               }
           },
       })
   </script>

父子组件数据传递之监听子组件事件

<!-- 
        子组件提交事件给父组件
        $emit(事件名称，传给事件函数的参数)
        需求：使用子组件事件放大列表的字号
        1、在父组件中新增一个fontSize数据属性
        2、写了一个内联样式，样式关联该数据
        3、在子组件内绑定了一个自定义事件，并且触发事件的是一个表达式

     -->
    <div id="app1">
        <item-one
            v-for="item in list"
            v-bind:xiaohua="item"
            :key="item.id"
            :style="{fontSize:fontSize + 'em'}"
            v-on:cuihua="change"
        ></item-one>
    </div>
    <script>
        Vue.component('item-one', {
            props: ["xiaohua",], // 可以定义多个prop
            template: `<div>
                <p>{{xiaohua.text}}</p>
                <button @click="$emit('cuihua', 0.1, 1)">点击放大字号</button>
            </div>`
        })

        let vm1 = new Vue({
            el: "#app1",
            data: {
                list: [
                    {id: 0, text: '榴莲'},
                    {id: 1, text: '香蕉'},
                    {id: 2, text: '哈密瓜'},
                    {id: 3, text: '西瓜'}
                ],
                fontSize: 1
            },
            methods: {
                change(num, times) {
                    this.fontSize += num*times
                }
            },
            
        })
    </script>

<div id="app1">
        <button
            v-for="tab in tabs"
            :key="tab"
            @click="change(tab)"
            v-bind:class="['tab-button', {active: currentTab === tab}]">
        {{tab}}
        </button>
        <!-- is特性绑定哪个组件就显示哪个组件 -->
        <component :is="currentTabComponent" class="tab"></component>
    </div>
<script>
        Vue.component('tab-home', {
            template: `<div>Home Page</div>`
        })
        Vue.component('tab-list', {
            template: `<div>List Page</div>`
        })
        Vue.component('tab-news', {
            template: `<div>News Page</div>`
        })

        let vm1 = new Vue({
            el: "#app1",
            data: {
                tabs: ['home', 'list', 'news'],
                currentTab: 'home'
            },
            computed: {
                currentTabComponent() {
                    return 'tab-' + this.currentTab
                }
            },
            methods: {
                change(val) {
                    this.currentTab = val
                }
            },
        })
    </script>
